\newcommand\drawover[2]{
  \draw[lineback] ($0.75*#1+0.25*#2$) to ($0.75*#2+0.25*#1$);
  \draw[grid] #1 to #2;
}

\newcommand\drawcontrasto[4]{

  \draw #1 -- node[fill=white,sloped]{\tiny #3} ($0.5*#1+0.5*#2$);
  \draw ($0.5*#1+0.5*#2$) -- node[fill=white,sloped]{\tiny #4} #2;
}
